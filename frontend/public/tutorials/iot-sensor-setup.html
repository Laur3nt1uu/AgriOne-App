Asamblarea Hardware

15 minute

Conecteaza componentele la ESP32 conform schemei electronice de mai jos. Verifica toate conexiunile de doua ori inainte de alimentare pentru a evita deteriorarea componentelor.

ESP32 DevKit V1
DHT22
VCC→3.3V
DATA→GPIO4
GND→GND
Soil Moisture
VCC→3.3V
AOUT→GPIO34
GND→GND
LDR + 10kΩ
Divizor tensiune
OUTPUT→GPIO35
GND→GND
⚡ Schema de conexiuni pentru senzori AgriOne IoT
  • DHT22: Pinul VCC la 3.3V, pinul GND la GND, pinul DATA la GPIO4 pe ESP32
  • Soil Moisture: VCC la 3.3V, GND la GND, AOUT (analog out) la GPIO34
  • LDR: Un terminal la 3.3V prin rezistor 10kΩ, celalalt terminal la GPIO35 si GND
  • Verifica toate conexiunile cu multimetrul inainte de a conecta ESP32 la alimentare
Foarte Important: Nu conecta nimic la 5V! ESP32 functioneaza exclusiv la 3.3V. Tensiunea de 5V poate distruge definitiv microcontrollerul si senzorii conectati.

Programarea Firmware

15 minute

Incarca firmware-ul AgriOne pe ESP32 folosind Arduino IDE. Codul trimite automat datele catre API-ul AgriOne la fiecare 5 minute si intra in modul sleep pentru economisirea bateriei.

C++ / Arduino IDE 
/*
 * AgriOne IoT Sensor Firmware v2.1
 * Compatible cu ESP32 DevKit V1 si senzori DHT22, Soil Moisture, LDR
 * Transmisie automata la fiecare 5 minute cu deep sleep
 */

#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>
#include <DHT.h>

// Configuratie retea WiFi
const char* ssid = "NUMELE_WIFI_TAU";
const char* password = "PAROLA_WIFI_TAU";

// API Key din AgriOne > Profil > Senzori
const char* apiKey = "agrione_xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx";
const char* sensorId = "sensor_001"; // ID unic pentru acest senzor

// Definire pini senzori conform schemei
#define DHT_PIN 4       // GPIO4 pentru DHT22
#define SOIL_PIN 34     // GPIO34 pentru soil moisture (analog)
#define LDR_PIN 35      // GPIO35 pentru LDR (analog)

// Initializare senzor DHT22
DHT dht(DHT_PIN, DHT22);

// Configurare deep sleep (5 minute = 300 secunde)
#define SLEEP_TIME_US 300000000ULL // 5 minute in microsecunde

void setup() {
  Serial.begin(115200);
  delay(1000);

  Serial.println("AgriOne IoT Sensor v2.1");
  Serial.println("Initializare senzori...");

  // Initializare senzori
  dht.begin();

  // Conectare la WiFi
  connectWiFi();

  // Citire si trimitere date
  readAndSendData();

  // Intrare in deep sleep pentru economisirea bateriei
  Serial.println("Intrare in sleep mode pentru 5 minute...");
  esp_sleep_enable_timer_wakeup(SLEEP_TIME_US);
  esp_deep_sleep_start();
}

void loop() {
  // Nu se executa niciodata datorita deep sleep
}

void connectWiFi() {
  WiFi.begin(ssid, password);
  Serial.print("Conectare la WiFi");

  int attempts = 0;
  while (WiFi.status() != WL_CONNECTED && attempts < 20) {
    delay(500);
    Serial.print(".");
    attempts++;
  }

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println();
    Serial.println("WiFi conectat!");
    Serial.print("IP: ");
    Serial.println(WiFi.localIP());
  } else {
    Serial.println("Eroare conectare WiFi!");
  }
}

void readAndSendData() {
  // Citire senzor DHT22
  float temperature = dht.readTemperature();
  float airHumidity = dht.readHumidity();

  // Citire senzor umiditate sol
  int soilRaw = analogRead(SOIL_PIN);
  float soilMoisture = map(soilRaw, 0, 4095, 100, 0); // Conversie la procent

  // Citire senzor lumina LDR
  int ldrRaw = analogRead(LDR_PIN);
  float lightLevel = map(ldrRaw, 0, 4095, 0, 100000); // Conversie la lux (aproximativ)

  // Afisare date in Serial Monitor pentru debug
  Serial.println("=== Date Senzori ===");
  Serial.print("Temperatura: "); Serial.print(temperature); Serial.println("°C");
  Serial.print("Umiditate aer: "); Serial.print(airHumidity); Serial.println("%");
  Serial.print("Umiditate sol: "); Serial.print(soilMoisture); Serial.println("%");
  Serial.print("Intensitate lumina: "); Serial.print(lightLevel); Serial.println(" lux");

  // Trimitere date la AgriOne API
  sendToAgriOne(temperature, airHumidity, soilMoisture, lightLevel);
}

void sendToAgriOne(float temp, float airHum, float soilMoist, float light) {
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    http.begin("https://api.agri-one.com/v1/sensors/readings");
    http.addHeader("Content-Type", "application/json");
    http.addHeader("Authorization", "Bearer " + String(apiKey));

    // Creare JSON payload
    DynamicJsonDocument doc(512);
    doc["sensor_id"] = sensorId;
    doc["temperature"] = temp;
    doc["air_humidity"] = airHum;
    doc["soil_moisture"] = soilMoist;
    doc["light_level"] = light;
    doc["battery_level"] = 85; // Placeholder pentru citire baterie
    doc["timestamp"] = millis();

    String payload;
    serializeJson(doc, payload);

    int httpCode = http.POST(payload);

    if (httpCode > 0) {
      String response = http.getString();
      Serial.print("HTTP Response: ");
      Serial.println(httpCode);
      Serial.println(response);
    } else {
      Serial.print("Eroare HTTP: ");
      Serial.println(httpCode);
    }

    http.end();
  } else {
    Serial.println("WiFi deconectat!");
  }
}
  • Instaleaza Arduino IDE si adauga board-ul ESP32 din Board Manager
  • Instaleaza librariile: DHT sensor library si ArduinoJson din Library Manager
  • Inlocuieste NUMELE_WIFI_TAU si PAROLA_WIFI_TAU cu datele retelei tale
  • Copiaza API Key din AgriOne > Profil > Senzori si introdu-l in cod
  • Selecteaza ESP32 Dev Module, portul corect si apasa Upload
Cod complet disponibil: Descarci codul complet cu toate functiile avansate de pe github.com/agrione/esp32-firmware. Include gestionare erori, calibrare automata si optimizari baterie.

Calibrarea si Testare

10 minute

Dupa instalarea firmware-ului, calibreaza senzorii pentru a obtine citiri precise si testeaza transmisia datelor in AgriOne dashboard. Acest proces asigura monitorizarea corecta a conditiilor.

Intrare in sleep mode pentru 5 minute...
  • Deschide Serial Monitor (115200 baud) pentru a verifica functionarea senzorului
  • Pentru calibrarea solului: plaseaza senzorul in sol uscat si noteaza valoarea minima
  • Plaseaza senzorul in sol saturat cu apa si noteaza valoarea maxima
  • In AgriOne > Senzori > Calibrare, introdu valorile min/max pentru mapare corecta
  • Verifica in Dashboard ca datele apar si sunt realiste (HTTP Response 200)
IMPORTANT: Recalibreaza senzorii de umiditate la schimbarea locatiei, tipului de sol sau sezonului pentru rezultate optime. Senzorii pot deriva in timp.
